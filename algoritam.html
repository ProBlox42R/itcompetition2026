Арабски към Римски (JavaScript)

Логиката използва итерация през речник от стойности, като изважда най-голямата възможна цифра от числото.

// Арабски -> Римски
function arabicToRoman(number) {
    let roman = "";
    const list = {M:1000, CM:900, D:500, CD:400, C:100, XC:90, L:50, XL:40, X:10, IX:9, V:5, IV:4, I:1};

    if (number < 1 || number > 3999) 
        return "Въведете число между 1 и 3999";

    for (let key in list) {
        while (number >= list[key]) {
            roman += key;
            number -= list[key];
        }
    }
    return roman;
}

Римски към Арабски (JavaScript)

Алгоритъмът проверява двойки символи (като IV или CM), за да определи дали трябва да събира или изважда стойностите.

// Римски -> Арабски
function romanToArabic(roman) {
    roman = roman.toUpperCase();
    const values = {I:1, V:5, X:10, L:50, C:100, D:500, M:1000};
    let result = 0;

    for (let i = 0; i < roman.length; i++) {
        let cur = values[roman[i]];
        let next = values[roman[i+1]];

        if (next > cur) {
            result += (next - cur);
            i++;
        } else {
            result += cur;
        }
    }
    return result;
}
